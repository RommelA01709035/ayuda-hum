  function evaluarDecision() {
    const tipoConflicto = document.getElementById("tipoConflicto").value;
    const distancia = parseInt(document.getElementById("distancia").value);
    const pib = parseInt(document.getElementById("pib").value);
    const alianza = parseInt(document.getElementById("alianza").value);
    const escalada = parseInt(document.getElementById("escalada").value);
    const congreso = parseInt(document.getElementById("congreso").value);
    const opinion = parseInt(document.getElementById("opinionPublica").value);
    const presion = parseInt(document.getElementById("presionPotencias").value);

    let resDiv = document.getElementById("resultado");
    let decisionTxt = document.getElementById("decisionTitulo");
    let advertencias = document.getElementById("advertencias");
    let senal = document.getElementById("senal");

    let militar = 0, humanitaria = 0, neutral = 0;
    advertencias.innerHTML = "";
    resDiv.className = "resultado";

    // Lógica Base
    if (pib > 1000) { militar += 3; humanitaria += 2; } else { neutral += 2; }
    if (distancia > 5000) neutral += 3;
    if (alianza === 1) militar += 6;

    // Lógica Nueva: Opinión Pública
    if (opinion === 1) { // En contra
        neutral += 5;
        militar -= 3;
    } else if (opinion === 3) { // A favor
        militar += 3;
        humanitaria += 3;
    }

    // Lógica Nueva: Presión de Superpotencias
    if (presion === 3) {
        militar += 5;
        humanitaria += 2;
        neutral -= 4;
    }

    // Contexto de conflicto
    if (tipoConflicto === "humanitaria") humanitaria += 5;
    if (tipoConflicto === "terrorismo") militar += 4;
    if (escalada === 3) {
        neutral += 6;
        militar -= 2;
    }

    // Actualizar UI
    document.getElementById("militarScore").innerText = Math.max(0, militar);
    document.getElementById("humanitariaScore").innerText = Math.max(0, humanitaria);
    document.getElementById("neutralScore").innerText = Math.max(0, neutral);

    if (militar >= humanitaria && militar >= neutral) {
      resDiv.classList.add("verde");
      senal.innerHTML = "🛡️";
      decisionTxt.innerText = "RECOMENDACIÓN: INTERVENCIÓN / AYUDA MILITAR";
    } else if (humanitaria >= neutral) {
      resDiv.classList.add("amarillo");
      senal.innerHTML = "📦";
      decisionTxt.innerText = "RECOMENDACIÓN: ASISTENCIA HUMANITARIA";
    } else {
      resDiv.classList.add("rojo");
      senal.innerHTML = "🏳️";
      decisionTxt.innerText = "RECOMENDACIÓN: NEUTRALIDAD ESTRICTA";
    }

    // Alertas Dinámicas
    if (opinion === 1 && militar > neutral) {
      advertencias.innerHTML += "<li>Riesgo de inestabilidad interna: La población se opone a la acción militar.</li>";
    }
    if (presion === 3 && neutral > militar) {
      advertencias.innerHTML += "<li>Riesgo diplomático: Mantener la neutralidad podría traer sanciones de superpotencias.</li>";
    }
    if (escalada === 3) {
      advertencias.innerHTML += "<li>¡PELIGRO!: El riesgo nuclear aconseja retirar cualquier apoyo directo.</li>";
    }

    resDiv.style.display = "block";
    resDiv.scrollIntoView({behavior: 'smooth'});
  }
</script>
